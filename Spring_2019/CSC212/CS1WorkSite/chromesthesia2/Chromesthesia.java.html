package chromesthesia2;

import java.util.Scanner;
import javax.swing.JOptionPane;
import javax.swing.SwingUtilities;
import painter.SPainter;

public class Chromesthesia {
    public static void main(String[] args) {
        SwingUtilities.invokeLater(new chromesthesia2.Chromesthesia.ThreadForGUI());

    }

    private static class ThreadForGUI implements Runnable {
        @Override

        public void run() {
            new chromesthesia2.Chromesthesia();

        }

    }

    public Chromesthesia() {
        interpreter();

    }
    // FEATURED VARIABLES

    private static SPainter miro;
    private static chromesthesia2.Pitch[] pitches;

    // THE INTERPRETER
    public static void interpreter() {

        initialization(); // miro and pitches

        while ( true ) {
            String input = getInput();

            if (input.equalsIgnoreCase("EXIT")) {
                break;

            } else {
                try {
                    playMelody(input, pitches);
                } catch (Exception ex) {
                    showErrorMessage(ex.toString());
                }
            }
        }
        cleanup(); // miro has to go
    }

// METHODS PERTAINING TO THE CHROMESTHETIC PITCHES

    private static chromesthesia2.Pitch[] establishPitches(SPainter painter) {
        chromesthesia2.Pitch[] pitches = new chromesthesia2.Pitch[21];

        chromesthesia2.Pitch pitchMiddleC = new chromesthesia2.Pitch("C",painter);
        pitches[0] = pitchMiddleC;
        chromesthesia2.Pitch pitchLowC = new chromesthesia2.Pitch("C,",painter);
        pitches[1] = pitchLowC;
        chromesthesia2.Pitch pitchHighC = new chromesthesia2.Pitch("c",painter);
        pitches[2] = pitchHighC;

        chromesthesia2.Pitch pitchMiddleD = new chromesthesia2.Pitch("D",painter);
        pitches[3] = pitchMiddleD;
        chromesthesia2.Pitch pitchLowD = new chromesthesia2.Pitch("D,",painter);
        pitches[4] = pitchLowD;
        chromesthesia2.Pitch pitchHighD = new chromesthesia2.Pitch("d",painter);
        pitches[5] = pitchHighD;

        chromesthesia2.Pitch pitchMiddleE = new chromesthesia2.Pitch("E",painter);
        pitches[6] = pitchMiddleE;
        chromesthesia2.Pitch pitchLowE = new chromesthesia2.Pitch("E,",painter);
        pitches[7] = pitchLowE;
        chromesthesia2.Pitch pitchHighE = new chromesthesia2.Pitch("e",painter);
        pitches[8] = pitchHighE;

        chromesthesia2.Pitch pitchMiddleF = new chromesthesia2.Pitch("F",painter);
        pitches[9] = pitchMiddleF;
        chromesthesia2.Pitch pitchLowF = new chromesthesia2.Pitch("F,",painter);
        pitches[10] = pitchLowF;
        chromesthesia2.Pitch pitchHighF = new chromesthesia2.Pitch("f",painter);
        pitches[11] = pitchHighF;

        chromesthesia2.Pitch pitchMiddleG = new chromesthesia2.Pitch("G",painter);
        pitches[12] = pitchMiddleG;
        chromesthesia2.Pitch pitchLowG = new chromesthesia2.Pitch("G,",painter);
        pitches[13] = pitchLowG;
        chromesthesia2.Pitch pitchHighG = new chromesthesia2.Pitch("g",painter);
        pitches[14] = pitchHighG;

        chromesthesia2.Pitch pitchMiddleA = new chromesthesia2.Pitch("A",painter);
        pitches[15] = pitchMiddleA;
        chromesthesia2.Pitch pitchLowA = new chromesthesia2.Pitch("A,",painter);
        pitches[16] = pitchLowA;
        chromesthesia2.Pitch pitchHighA = new chromesthesia2.Pitch("a",painter);
        pitches[17] = pitchHighA;

        chromesthesia2.Pitch pitchMiddleB = new chromesthesia2.Pitch("B",painter);
        pitches[18] = pitchMiddleB;
        chromesthesia2.Pitch pitchLowB = new chromesthesia2.Pitch("B,",painter);
        pitches[19] = pitchLowB;
        chromesthesia2.Pitch pitchHighB = new chromesthesia2.Pitch("b",painter);
        pitches[20] = pitchHighB;

        return pitches;
    }

    private static chromesthesia2.Pitch find(String token, chromesthesia2.Pitch[] pitches) throws Exception {
        for ( int i = 0; i < pitches.length; i = i + 1 ) {

            chromesthesia2.Pitch pitch = pitches[i];
            if ( pitch.abcName().equals(token) ) {

                return pitch;
            }

        }

        throw new Exception("### PITCH " + token + " NOT FOUND");
    }

    private static void display(chromesthesia2.Pitch[] pitches) {
        for ( int i = 0; i < pitches.length; i = i + 1 ) {

            System.out.println(pitches[i].toString());
        }

    }

    private static void playMelody(String input, Pitch[] pitches) throws Exception {
        Scanner scanner = new Scanner(input);
        while ( scanner.hasNext() ) {
            String token = scanner.next();
            String pitchName;
            String duration = "";
            if ( token.indexOf(",") < 0 ) {
                pitchName = token.substring(0,1);
                duration = token.substring(1);
            } else {
                pitchName = token.substring(0,2);
                duration = token.substring(2);
            }
            if ( duration.length() == 0 ) { duration = "1"; }
            Pitch pitch = find(pitchName,pitches);
            pitch.play(duration);
        }
    }

    // INITIALIZATION, CLEANUP, GETTING INPUT, ERROR MESSAGING

    static private void showErrorMessage(String message) {
        miro.setVisible(false);
        JOptionPane.showMessageDialog(null, message);

    }

    private static void initialization() {
        // ESTABLISH THE PAINTER AND GIVE IT A SUBSTANTIAL BRUSH WIDTH
        miro = new SPainter("Chromesthesia",500,500);
        miro.setVisible(false);
        miro.setBrushWidth(7);
        // ESTABLISH THE CHROMESTITIC PITCH CLASS OBJECTS
        pitches = establishPitches(miro);
        display(pitches);

    }

    private static String getInput() {
        miro.setVisible(false);
        String label = "Please enter a melody in ABC notation, or EXIT ...     ";
        String input = JOptionPane.showInputDialog(null,label);
        miro.setVisible(true);
        if ( input == null ) { input = ""; }
        return input;

    }

    static private void cleanup() {
        System.exit(0);

    }

}
