/* 
 *Model a die in terms of two properties: 
 * - order, the number of faces 
 * - top, the value of the top face 
 */

package chance;

public class Die {
    //THE INSTANCE VARIABLES (STATE)
    private int order;
    private int top;

    //THE CONSTRUCTORS
    public Die() {
        order = 6;
        top = (int) ((Math.random()* 6) + 1);
    }
    //THE METHODS (BEHAVIOR)
    public int top() {
        return top;
    }
    public void roll(){
        top = (int) ((Math.random() * order) + 1);
    }
}
