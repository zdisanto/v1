package balloonpeople;

import painter.SPainter;
import shapes.SCircle;
import java.awt.*;
import java.util.Random;

public class BalloonPerson {
    private String name;
    private int age;
    private double height;
    private double headSize = 50;
    private double shrinkFactor = 1.5;
    private SCircle base = new SCircle(headSize);


    public BalloonPerson(String name, int age, double height) {
        this.name = name;
        this.age = age;
        this.height = height;
    }

    public String toString() {
        return " Name: " + name + " Age: " + age + " Height: " + height + "\n";
    }

    public void paintPerson(SPainter painter) {
        randomColor(painter);
        head(painter);
        neckDown(painter);
        legs(painter);
    }

    private void legs(SPainter painter) {
        painter.mbk(base.radius() + (height * 1/2));
        //Legs (r)
        painter.mrt(.2*base.radius());
        painter.mrt(.2*base.radius());
        painter.mrt(.2*base.radius());
        painter.mrt(.2*base.radius());
        painter.paint(base);
        painter.mlt(.2*base.radius());
        painter.mlt(.2*base.radius());
        painter.mlt(.2*base.radius());
        painter.mlt(.2*base.radius());
        // (l)
        painter.mlt(.2*base.radius());
        painter.mlt(.2*base.radius());
        painter.mlt(.2*base.radius());
        painter.mlt(.2*base.radius());
        painter.paint(base);
        painter.mrt(.2*base.radius());
        painter.mrt(.2*base.radius());
        painter.mrt(.2*base.radius());
        painter.mrt(.2*base.radius());

        //Move to Center
        painter.mfd(base.radius() + (height * 1/2));
        painter.mfd(base.radius() + (height * 1/2));
    }

    private void neckDown(SPainter painter) {
        //Body
        painter.mbk(base.radius() + (height * 2/3));
        base.shrink(headSize / shrinkFactor);
        painter.paint(base);

        //Arms (r)
        base.expand(height / (shrinkFactor * shrinkFactor));
        painter.mrt(base.diameter());
        painter.paint(base);
        painter.mlt(base.diameter());
        //(l)
        painter.mlt(base.diameter());
        painter.paint(base);
        painter.mrt(base.diameter());
    }

    private void head(SPainter painter) {
        //Head
        painter.paint(base);

    }
    private void randomColor(SPainter painter) {
        Random rgen = new Random();
        int r = rgen.nextInt(255);
        int g = rgen.nextInt(255);
        int b = rgen.nextInt(255);
        painter.setColor(new Color(r, g, b));
    }
    public void increment(SPainter painter) {
        painter.mrt(base.diameter()*4);
    }
}
