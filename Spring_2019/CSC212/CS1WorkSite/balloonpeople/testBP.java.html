package balloonpeople;

import painter.SPainter;
import javax.swing.*;
import java.awt.geom.Point2D;

public class testBP {
    public testBP(){
        SPainter painter = new SPainter("Balloon Person Tester", 400, 400);

        int startingX = 200;
        int startingY = 200;
        painter.moveTo(new Point2D.Double(startingX, startingY));

        BalloonPerson person = new BalloonPerson("Test",20, 50);
        person.paintPerson(painter);
        System.out.print(person);
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                new testBP();
            }
        });
    }
}
