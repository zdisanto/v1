package npw;

import painter.SPainter;
import shapes.SSquare;

import javax.swing.*;
import java.awt.*;
import java.util.Random;
import java.util.Scanner;

public class Stella {

    public Stella() {
        paintTheImage();
    }

    private void paintTheImage() {
        //Variables
        int Canvas = 800;
        int lSquare = (Canvas -100);
        // Canvas and Prompt
        int numOfSquares = getNumb("How many Squares on Stella");
        SPainter p = new SPainter("Stella", Canvas, Canvas);
        //The square
        SSquare sq = new SSquare(lSquare);
        //Change Var
        int sSide = lSquare / numOfSquares;

        CreateStella(p, numOfSquares, sSide, sq);
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                new Stella();
            }
        });
    }

    private void CreateStella(SPainter p, int numOfSquares, int sidechange, SSquare sq) {
        boolean aColor = true;
        Color rcolor1 = randomColor();
        Color rcolor2 = randomColor();

        for (int i = 0; i < numOfSquares; i++ ) {
            if (aColor == true ) {
                p.setColor( rcolor1);
                p.paint( sq );
                sq.shrink( sidechange );
                aColor = false;
            } else {
                p.setColor(rcolor2);
                p.paint(sq);
                sq.shrink(sidechange);
                aColor = true;
            }
        }
    }

    private static int getNumb(String prompt) {
        String nss = JOptionPane.showInputDialog(null,prompt+"?");
        Scanner scanner = new Scanner(nss);
        return scanner.nextInt();
    }

    private Color randomColor() {
        Random rgen = new Random();
        int r = rgen.nextInt(255);
        int g = rgen.nextInt(255);
        int b = rgen.nextInt(255);
        return new Color(r, g, b);
    }
}
