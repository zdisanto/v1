package npw;

import java.awt.Color;
import java.util.Random;
import javax.swing.SwingUtilities;
import painter.SPainter;
import shapes.SRectangle;

public class Invention2 {
    //REQUIRED INFRASTRUCTURE
    public Invention2() {
        paintTheImage();
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                new Invention2();
            }
        });
    }

    // THE PAINTER DOING ITS THING
    private void paintTheImage() {
        SPainter painter = new SPainter("Invention 2", 600, 600);
        int nrOfRectangles = 300;
        paintRectangles(painter, nrOfRectangles);  // ask IntelliJ to generate the stub
    }

    private void paintRectangles(SPainter painter, int nrOfRectangles) {
        int i = 1;
        while (i <= nrOfRectangles) {
            paintOneRectangle(painter);  // ask IntelliJ to generate the stub
            i = i + 1;
        }
    }

    private void paintOneRectangle(SPainter painter) {
        Random rgen = new Random();
        int rn = rgen.nextInt(6);
        if (rn == 0) {
            painter.setColor(new Color(255, 83, 13 ));
        }
        else if (rn == 1) {
            painter.setColor(new Color(41, 164, 255 ));
        }
        else if (rn == 2) {
            painter.setColor(new Color(255, 13, 255 ));
        }
        else if (rn == 3) {
            painter.setColor(new Color(28, 255, 230 ));
        }
        else if (rn == 4) {
            painter.setColor(new Color(255, 250, 88 ));
        }
        else {
            int r = rgen.nextInt(255);
            int b = rgen.nextInt(255);
            int y = rgen.nextInt(255);
            painter.setColor(new Color(r,b,y));
        }
        painter.move();
        int recSide = 30;
        SRectangle rec = new SRectangle(recSide, recSide*2);
        painter.paint(rec);
        painter.setColor(Color.BLACK);
        painter.draw(rec);
    }

}
