package shapes;

import java.util.Scanner;

public class ShippingContainer {
    public static void main(String[] args) {
        //In
        Scanner scanner = new Scanner(System.in);
        System.out.println("Width = ");
        double width = scanner.nextDouble();
        System.out.println("Length = ");
        double length = scanner.nextDouble();
        System.out.println("Height = ");
        double height = scanner.nextDouble();
        //Objects
        SRectangle face = new SRectangle(width, length);
        SRectangle key = new SRectangle(height, face.diagonal());
        double distance = key.diagonal();
        //Prints
        System.out.println("Area of Face: " + face);
        System.out.println("Area of Key: " + key);
        System.out.println("Length of Distance from Corner to Corner: " + distance);
    }
}
