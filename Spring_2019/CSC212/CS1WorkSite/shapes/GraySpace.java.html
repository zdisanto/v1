package shapes;

public class GraySpace {
    public static void main(String[] args) {
        //Variables (gsg = gray square gap, osg = orange square gap)
        double gsg = 3;
        double osg = 4.5;
        double topGap = 11.5;

        //Lengths (gsg = gray square gap, lgc= light gray circle)
        double darkgraySide = 40;
        double lgcRadius = (darkgraySide - (gsg * 2)) /4;
        double ocRadius = (darkgraySide - (topGap + osg)) /2;

        //# of Objects
        double numberDarkGray = 1;
        double numberLightGray = 2;
        double numberOrange = 1;

        //Circles (oc = orange circle)
        SCircle lgc = new SCircle(lgcRadius);
        SCircle oc =  new SCircle(ocRadius);

        //Some Math using shapes ;)
        SSquare DarkGray = new SSquare(darkgraySide);
        SSquare LightGray = lgc.inscribingSquare();
        SSquare orangeSquare = oc.inscribingSquare();

        //Area
        double areaOfDarkGray = DarkGray.area();
        double areaOfLightGray = LightGray.area();
        double areaOfOrange = orangeSquare.area();

        //Totals
        double totalDarkGray = areaOfDarkGray*numberDarkGray;
        double totalLightGray = areaOfLightGray*numberLightGray;
        double totalOrange = areaOfOrange*numberOrange;

        //Final Maths
        double finalareaDarkGray = areaOfDarkGray-(totalLightGray+totalOrange);

        //Print
        System.out.println("Area of Dark Gray Sqaure: " + totalDarkGray);
        System.out.println("Area of a Light Square: " + totalLightGray);
        System.out.println("Area of Orange Square: " + totalOrange);
        System.out.println("Area of Remaining Dark Gray Square: " + finalareaDarkGray);
    }

}
