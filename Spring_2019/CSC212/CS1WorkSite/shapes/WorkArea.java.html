package shapes;

public class WorkArea {
    public static void main(String[] args) {
        //Desk (1)
        SRectangle desk = new SRectangle(24, 36);
        System.out.println("Desk = " + desk.toString());
        System.out.println("Area of desk = " + desk.area());
        double deskArea = desk.area();

        //Books (2)
        SRectangle book = new SRectangle(11, 8.5);
        System.out.println("Book = " + book.toString());
        System.out.println("Area of book = " + book.area());
        double areaOfBooks = (book.area() * 2);
        System.out.println("Total area of two books = " + areaOfBooks);

        //Glass (3)
        SCircle glass = new SCircle(1.15);
        System.out.println("Glass = " + glass.toString());
        System.out.println("Area of glass = " + glass.area());

        //Coaster (3)
        SSquare coaster = glass.circumscribingSquare();
        System.out.println("Coaster = " + coaster.toString());
        System.out.println("Area of coaster = " + coaster.area());
        double areaOfCoasters = (coaster.area() * 3);
        System.out.println("Total area of three coasters = " + areaOfCoasters);

        //Plate (3)
        SCircle plate = new SCircle(6);
        System.out.println("Plate = " + plate.toString());
        System.out.println("Area of plate = " + plate.area());
        double areaOfPlates = (plate.area() * 3);
        System.out.println("Total area of three plates = " + areaOfPlates);

        //All Objects
        double allObjectAreas = (areaOfPlates + areaOfCoasters + areaOfBooks);
        System.out.println("Total area of all objects on desk = " + allObjectAreas);

        //Remaining Areas
        double remainingArea = (deskArea - allObjectAreas);
        System.out.println("The remaining area of desk = " + remainingArea);

    }

}
