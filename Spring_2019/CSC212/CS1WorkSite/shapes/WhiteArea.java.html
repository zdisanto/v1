package shapes;

public class WhiteArea {
    public static void main(String[] args) {
        //Doubles
        double dotDiameter = .75 / 8;
        double sideLength = .75;
        double numberOfFaces = 6;
        double numberOfDots = 21;

        //S's
        SSquare face = new SSquare(sideLength);
        SCircle dot = new SCircle(dotDiameter / 2);

        //Areas
        double areaOfFace = face.area();
        double areaOfDot = dot.area();

        //Total Areas
        double sumOfFaces= areaOfFace*numberOfFaces;
        double sumOfDots = areaOfDot*numberOfDots;

        //Remaining
        double remainingWhiteArea = sumOfFaces - sumOfDots;

        //Print
        System.out.println("Total area of the die " + sumOfFaces);
        System.out.println("Total area of remaining white area " + remainingWhiteArea);
    }
}
