private boolean isNeighbor(Tile s, Tile t) {
if (s.row == t.row && (s.col == t.col - 1 || s.col == t.col + 1)) {

}
else if (s.col == t.col && (s.row == t.row - 1 || s.row == t.row + 1)){
return true;
}
else if ((s.row == t.row + 1 || s.row == t.row - 1) && (s.col == t.col + 1 || s.col == t.col - 1) ||
(s.col == t.col + 1 || s.col == t.col - 1) && (s.row == t.row + 1 || s.row == t.row - 1)) {

} else {
return false;
}
